h1 {

	text-align: center;
}

h2 {
	color: pink;
	text-align: center;
}

#heading {
	background-color: blue;
}

.square {
	width: 100px;
	height: 100px;
	background-color: red;
}



.rect {
	width: 50px;
	height: 100px;
	background-color: blue;
}

.tri {
	width: 0px;
	height: 0px;
	border-right: 50px solid yellow;
	border-left: 50px solid orange;
	border-top: 50px solid purple;
	border-bottom: 50px solid green;
}

.circle {
	width: 100px;
	height: 100px;
	background-color: yellow;
	border-radius: 100px;
}

#movingCircle{
	position: relative;
	
	animation: path 5s infinite;
}

@keyframes path {
	0% {
		left: 0px;
		top: 0px;
	}
	25% {
		left: 100px;
		top: 0px;
	}
	50% {
		left: 100px;
		top: 100px;
	}
	75% {
		left: 0px;
		top: 100px;
	}
	100% {
		left: 0px;
		top: 0px;
	}
}



#tilted {
	transform: rotate(30deg);
	position: relative;
	
	animation-name: triPath;
	animation-duration: 5s;
}

@keyframes triPath {
	0% {
		left: 0px;
		top: 0px;
	}
	33% {
		left: 100px;
		top: 100px;
	}
	66% {
		left: 0px;
		top: 100px;
	}
	100% {
		left: 0px;
		top: 0px;
	}
}

#movingSquare {
	position: relative;
	
	animation-name: move;
	animation-duration: 2s;
}

@keyframes move {
	from {
		left: 0px;
		background-color: red;
	}
	to {
		left: 100px;
		background-color: green;
	}
}

#movingTriangle {
	position: relative;
	
	animation-name: triangleMove;
	animation-duration: 2s;
	animation-iteration-count: infinite;
	animation-direction: alternate;
}

@keyframes triangleMove {
	from {
		left: 0px;
		transform: rotate(0deg);
	}
	to {
		left: 1000px;
		transform: rotate(360deg);
	}
}
