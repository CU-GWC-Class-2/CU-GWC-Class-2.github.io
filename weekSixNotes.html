function changeColor(name) {
        var r = Math.random()*256;
        var g = Math.random()*256;
        var b = Math.random()*256;
        //"hello" + "there" --> "hellothere"
        
        var color = "rgb("+r+","+g+","+b+")";
        
        document.getElementById(name).style.backgroundColor = color;
}
