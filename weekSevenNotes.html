function hello(){

		alert(word[i]);

	}
}

function changeImage()
{
    // list of image urls
    var img_list  = ["http://www.spirit-animals.com/wp-content/uploads/2013/10/Panda-2-sm.jpg", "https://static.pexels.com/photos/104827/cat-pet-animal-domestic-104827.jpeg", "https://www.what-dog.net/Images/faces2/scroll001.jpg"];

    // get the url of the image currently on the page
    var img_curr = document.getElementById("animalImage").src;

    // loop through list of image urls
    for (var i=0; i < img_list.length; i=i+1) 
    {
        var img = img_list[i]; // set variable img equal to element at index i of list of images
        if (img == img_curr) // test to see if you've found the current image
        {
            if (i == img_list.length - 1) // found the last image in the list
            { 
                document.getElementById("animalImage").src = img_list[0]; // loop back to first image
            }
            else // not the last image in the list
            { 
                document.getElementById("animalImage").src = img_list[i + 1]; // change to next image in the list
            }
        }
    }   
}
